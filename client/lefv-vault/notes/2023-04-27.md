```yaml

name: Automerge Pull Requests

on:
  pull_request:
    types: [closed]

jobs:
  automerge:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Merge pull request into develop
        if: github.event.pull_request.merged && github.head_ref == 'release/*'
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          git config user.name "Automerge Bot"
          git config user.email "automerge-bot@example.com"
          git fetch --no-tags origin develop
          git merge --no-ff origin/${{ github.head_ref }} -m "Automerge ${{ github.head_ref }} into develop" && git push origin develop


```
